\item 2x Frame vertex
\item 4x Frame vertex with foot
\item 8x Bushing
\item 2x Pulley
\item 2x Rod clamp
\item 1x X carriage
\item 1x X end idler
\item 1x X end motor
\item 1x Y motor bracket

\item 68x M8 washer
\item 2x M8 spring
\item 19x M3 nut
\item 25x M3 washer
\item 17x M3 10mm screw
\item 3x M3 20mm screw
\item 6x M3 25mm screw
\end{itemize}
